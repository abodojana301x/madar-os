typedef struct UIElement {

🎨 4. الرسم (Rendering System)
void render_element(UIElement* el) {

    if (!el->visible) return;

    el->render(el);
}

void render_container(UIContainer* c) {

    render_element((UIElement*)c);

    for (int i = 0; i < c->child_count; i++) {
        render_element(c->children[i]);
    }
}

🔘 5. Button Widget
typedef struct {
    UIElement base;

    char* text;
    int is_hovered;

} Button;

void render_button(UIElement* el) {

    Button* btn = (Button*)el;

    int color = btn->is_hovered ? COLOR_BTN_HOVER : COLOR_BTN;

    draw_rect(el->x, el->y, el->width, el->height, color);
    draw_text(el->x + 5, el->y + 15, btn->text);
}

📐 7. Layout بسيط (Vertical / Horizontal)
🎯 بدون تعقيد… بس ترتيب عناصر
void layout_vertical(UIContainer* c) {

    int offset = 0;

    for (int i = 0; i < c->child_count; i++) {

        UIElement* child = c->children[i];

        child->y = c->base.y + offset;
        child->x = c->base.x;

        offset += child->height + 5;
    }
}

🖱️ 8. Event Handling داخل UI
🎯 تحديد العنصر اللي اتضغط عليه
UIElement* find_clicked(UIContainer* c, int x, int y) {

    for (int i = c->child_count - 1; i >= 0; i--) {

        UIElement* el = c->children[i];

        if (x >= el->x && x <= el->x + el->width &&
            y >= el->y && y <= el->y + el->height) {

            return el;
        }
    }

    return NULL;
}

🖱️ التنفيذ:
void handle_ui_click(UIContainer* root, int x, int y) {

    UIElement* el = find_clicked(root, x, y);

    if (el && el->on_click) {
        el->on_click(el);
    }
}
