🧱 1. تعريف النافذة (Window Structure)
typedef struct {
    int id;

    int x, y;
    int width, height;

    int is_focused;
    int is_dragging;

} Window;

📦 2. إدارة النوافذ (Window List)
#define MAX_WINDOWS 10

Window windows[MAX_WINDOWS];
int window_count = 0;
int focused_window = -1;

➕ 3. إنشاء نافذة
int create_window(int x, int y, int w, int h) {

    if (window_count >= MAX_WINDOWS) return -1;

    Window* win = &windows[window_count];

    win->id = window_count;
    win->x = x;
    win->y = y;
    win->width = w;
    win->height = h;
    win->is_focused = 1;
    win->is_dragging = 0;

    focused_window = win->id;

    window_count++;
    return win->id;
}

🎨 4. رسم النوافذ (Rendering)
🎯 الفكرة:
ارسم كل النوافذ
آخر نافذة = فوق
void render_windows() {
    for (int i = 0; i < window_count; i++) {

        Window* w = &windows[i];

        draw_rect(w->x, w->y, w->width, w->height, COLOR_WINDOW);

        // title bar
        draw_rect(w->x, w->y, w->width, 20, COLOR_PANEL);

        // title
        draw_text(w->x + 5, w->y + 15, "App");

        // close button
        draw_rect(w->x + w->width - 20, w->y, 20, 20, 0xFF4444);
    }
}

🖱️ 5. Focus System
🎯 عند الضغط:
حدد النافذة
خليها آخر واحدة في القائمة (top)
void focus_window(int id) {

    Window temp = windows[id];

    for (int i = id; i < window_count - 1; i++) {
        windows[i] = windows[i + 1];
    }

    windows[window_count - 1] = temp;

    focused_window = window_count - 1;
}

🖐️ 6. تحريك النافذة (Drag System)
🎯 يبدأ لما تضغط على الـ title bar
int drag_offset_x, drag_offset_y;

void handle_mouse_down(int x, int y) {

    for (int i = window_count - 1; i >= 0; i--) {

        Window* w = &windows[i];

        // check title bar
        if (x >= w->x && x <= w->x + w->width &&
            y >= w->y && y <= w->y + 20) {

            focus_window(i);

            w = &windows[window_count - 1];

            w->is_dragging = 1;

            drag_offset_x = x - w->x;
            drag_offset_y = y - w->y;

            return;
        }
    }
}

🎯 أثناء التحريك
void handle_mouse_move(int x, int y) {

    if (focused_window == -1) return;

    Window* w = &windows[focused_window];

    if (w->is_dragging) {
        w->x = x - drag_offset_x;
        w->y = y - drag_offset_y;
    }
}

🎯 عند الإفلات
void handle_mouse_up() {

    if (focused_window == -1) return;

    windows[focused_window].is_dragging = 0;
}

❌ 7. غلق النافذة (Close Window)
void close_window(int id) {

    for (int i = id; i < window_count - 1; i++) {
        windows[i] = windows[i + 1];
    }

    window_count--;

    focused_window = window_count - 1;
}

🖱️ الضغط على زر الإغلاق:

if (x >= w->x + w->width - 20 && y <= w->y + 20) {
