    EVENT_NONE,

    EVENT_MOUSE_MOVE,
    EVENT_MOUSE_DOWN,

    EVENT_KEY_DOWN,
    EVENT_KEY_UP

typedef struct {

    int mouse_x;
    int mouse_y;

    int key;
} Event;

🔄 3. Event Queue (طابور الأحداث)
#define MAX_EVENTS 64

Event event_queue[MAX_EVENTS];
int event_count = 0;

void push_event(Event e) {
    if (event_count < MAX_EVENTS) {
        event_queue[event_count++] = e;
    }
}

void poll_input() {

    // مثال بسيط (pseudo)
    if (mouse_moved) {
        Event e = {EVENT_MOUSE_MOVE, mouse_x, mouse_y};
        push_event(e);
    }

    if (mouse_clicked) {
        Event e = {EVENT_MOUSE_DOWN, mouse_x, mouse_y};
        push_event(e);
    }

    if (key_pressed) {
        Event e = {EVENT_KEY_DOWN, 0, 0, key_code};
        push_event(e);
    }
}

🧭 5. توزيع الأحداث (Event Dispatching)
void process_events() {

    for (int i = 0; i < event_count; i++) {

        Event* e = &event_queue[i];

        switch (e->type) {

            case EVENT_MOUSE_DOWN:
                handle_mouse_down(e->mouse_x, e->mouse_y);
                break;

            case EVENT_MOUSE_MOVE:
                handle_mouse_move(e->mouse_x, e->mouse_y);
                break;

            case EVENT_MOUSE_UP:
                handle_mouse_up();
                break;

            case EVENT_KEY_DOWN:
                handle_key_down(e->key);
                break;
        }
    }

    event_count = 0;
}

void cycle_windows() {

    if (window_count <= 1) return;

    Window last = windows[window_count - 1];

    for (int i = window_count - 1; i > 0; i--) {
        windows[i] = windows[i - 1];
    }

    windows[0] = last;

    focused_window = window_count - 1;
}

🖱️ 9. تحسين الماوس (Hover Detection)

🎯 لإضافة إحساس احترافي:

int hovered_window = -1;

void update_hover(int x, int y) {

    hovered_window = -1;

    for (int i = window_count - 1; i >= 0; i--) {
        Window* w = &windows[i];

        if (x >= w->x && x <= w->x + w->width &&
            y >= w->y && y <= w->y + w->height) {

            hovered_window = i;
            return;
        }
    }
}
